addButton: function() { this.$.force.addNode({ id: 'b', title: 'Title B', year: 2001 }); this.selected = 0; },
updateButton: function() { var id = this.$.force.getNodeId("Title B"); var node = { id: id, title: "Title BBB", year: 2004 }; this.$.force.updateNode(node); this.selected = 1; },
deleteButton: function() { var node = this.$.force.nodes.filter(function(n){return n.title == "Title B"})[0]; if (typeof(node) == "object") { console.log("delete node of '" + node.id + "'"); this.$.force.deleteNode(node.id); } this.selected = 2; },
addLinkButton: function() { this.$.force.addLink("c", "a"); this.selected = 3; },
deleteLinkButton: function() { this.$.force.deleteLink("c", "a"); this.selected = 4; },
